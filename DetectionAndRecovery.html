patient_age_dict = {'patient_age_data':[]}
for name,group in df_patient.groupby('agebracket'):
    total_cases = group.shape[0]
    recovered_cases = group[group['finalstatus']=='Recovered'].shape[0]
    deceased_cases = group[group['finalstatus']=='Deceased'].shape[0]
    percentage_recovered = (recovered_cases / total_cases) * 100
    percentage_deceased = (deceased_cases / total_cases) * 100
    patient_age_dict['patient_age_data'].append({
        'age': float(name), \
        'totalcases': total_cases, \
        'recoveredcases': recovered_cases, \
        'deceasedcases': deceased_cases, \
        'percentagerecovered': percentage_recovered, \
        'percentagedeceased': percentage_deceased, \
        'meanrecovereddduration': group[group['finalstatus']=='Recovered']['diseaseduration'].mean(), \
        'meandeceasedduration': group[group['finalstatus']=='Deceased']['diseaseduration'].mean(),
    })
