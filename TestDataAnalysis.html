Plot bar graph denoting total samples tested by each state

In [15]:
def plot_state_vs_sample_tested():
    label = df_total_tests['state']
    index = np.arange(len(label))
    plt.figure(figsize=(15,10))
    plt.bar(index, df_total_tests['sampletested'])
    plt.xlabel('State', fontsize=20)
    plt.ylabel('Sample tested in last 10 days', fontsize=20)
    plt.xticks(index, label, fontsize=5, rotation=85, size=10)
    plt.title('State vs Samples tested by each state in last 10 days')
    plt.show()
plot_state_vs_sample_tested()
